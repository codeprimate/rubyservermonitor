                def test_port(domain,port)
                        msg = " * Testing #{domain}:#{port} => "
                        now = Time.now
                        result = get_port(domain,port)
                        elapsed = Time.now - now
                        msg += "#{result ? 'OK' : 'Failed!'} (#{elapsed}s)  [#{Time.now}]"
                        @log << msg
                        [result,elapsed]
                end
