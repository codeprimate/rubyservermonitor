                def test_url(url)
                        msg =  " * Testing URL #{url} => "
                        now = Time.now
                        result = get_url(url)
                        elapsed = Time.now - now
                        msg += "#{result ? 'OK' : 'Failed!'} (#{elapsed}s)  [#{Time.now}]"
                        @log << msg
                        [result,elapsed]
                end
